\begin{frame}{Research proposal: selection signals in MSME funding}
  \begin{center}
  \Large
  \highlight{Do the signals used to select firms generate the\\ value public funding is meant to create?}
  \end{center}
  \vspace{0.35cm}
  \normalsize
  \begin{itemize}
    \item Which observable signals enter the funding decision?
    \item Do these signals predict innovation, spillovers, and economic value?
    \item Do programmes weight signals differently when they define value differently?
  \end{itemize}
  \vspace{0.3cm}
  \small
  The project separates the \highlight{selection mechanism} from the grant itself.
\end{frame}

\begin{frame}{Theoretical framework: selection signals as policy instruments}
  \large
  The project treats selection signals as \highlight{proxies for expected public value}.
  \vspace{0.45cm}
  \begin{center}
  \large
  Policy objective $\rightarrow$ Selection signals $\rightarrow$ Funding decision $\rightarrow$ Outcome layers
  \end{center}
  \vspace{0.45cm}
  \normalsize
  \begin{itemize}
    \item Programmes define value differently: innovation, spillovers, or local economic value.
    \item Granting authorities translate these objectives into signal weights.
    \item The empirical question is whether these weights predict the outcomes they are meant to generate.
  \end{itemize}
  \vspace{0.25cm}
  \small
  The grant is therefore not the only treatment; the selection rule is part of the mechanism.
\end{frame}

  \end{tikzpicture}
  \end{center}

  The unit of analysis is the \highlight{firm--application}. The project links three layers of data:
  \vspace{0.15cm}
  \begin{itemize} \small
    \item \highlight{Selection signals:} founder, project, technology, and regional characteristics
    \item \highlight{Selection rule:} how programmes weight these signals in the funding decision
    \item \highlight{Outcome layers:} innovation, spillover potential, and economic value
  \end{itemize}
  \vspace{0.15cm}

  \vspace{0.15cm}
  \small The empirical object is not the grant alone, but the link between \highlight{what programmes select for} and \highlight{what selected firms later generate}.
\end{frame}


\begin{frame}{Identification strategy}
  \large Identification comes from variation in \highlight{selection rules}, not only from grant receipt.
  \vspace{0.35cm}
  \normalsize The project compares how different programmes weight observable signals and whether those weights predict later outcomes.
  \vspace{0.25cm}
  \begin{itemize}
    \item Compare applicants within programmes and calls
    \item Use rankings, thresholds, or score cutoffs where available
    \item Compare programmes that define public value differently
    \item Test alignment between signal weights and realised outcome layers
  \end{itemize}
  \vspace{0.35cm}
  \small Where clean cutoffs exist, the design can estimate local causal effects. Otherwise, the core estimate is the \highlight{selection--outcome alignment gap}.
\end{frame}

\begin{frame}{Expected Contribution} 
  \begin{itemize} 
    \item \textbf{Empirical:} estimate whether selection signals generate the outcomes used to justify public support. 
    \item \textbf{Conceptual:} separate the effect of receiving a grant from the design of the rules that allocate grants. 
    \item \textbf{Policy-relevant:} identify where programme objectives and actual selection practice diverge.
    \item \textbf{PhD logic:} the project can begin as one empirical paper on grant selection design and develop into a broader research agenda on innovation policy. 
  \end{itemize} 
\end{frame}
